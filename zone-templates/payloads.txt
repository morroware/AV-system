; power=sendir,1:1,1,58000,1,1,192,192,48,145,48,145,48,48,48,145,48,145,...
; guide=sendir,1:1,1,58000,1,1,192,192,48,145,48,145,48,48,48,145,48,145,...
;
; You can capture IR codes using a learning remote or IR capture device
